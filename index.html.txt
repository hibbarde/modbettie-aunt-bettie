    function App() {
      const [messages, setMessages] = useState([{ role: "assistant", content: WELCOME }]);
      const [input, setInput] = useState("");
      const [loading, setLoading] = useState(false);
      const [entered, setEntered] = useState(false);
      const [currentModule, setCurrentModule] = useState(1);
      const [showResults, setShowResults] = useState(false);
      const [showPhotog, setShowPhotog] = useState(false);
      const [summaryText, setSummaryText] = useState("");
      const [usedChoices, setUsedChoices] = useState(new Set());
      const bottomRef = useRef(null);

      useEffect(() => { bottomRef.current?.scrollIntoView({ behavior: "smooth" }); }, [messages, loading]);

      const sendMessage = async (text) => {
        if (!text.trim() || loading) return;
        const updated = [...messages, { role: "user", content: text }];
        setMessages(updated);
        setInput("");
        setLoading(true);
        try {
          const res = await fetch("https://api.anthropic.com/v1/messages", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({
              model: "claude-sonnet-4-20250514",
              max_tokens: 1500,
              system: SYSTEM_PROMPT,
              messages: updated.map(m => ({ role: m.role, content: stripModuleTag(m.content).replace(/\[CHOICES:[^\]]+\]/g, "") }))
            })
          });
          const data = await res.json();
          const reply = data.content?.[0]?.text || "Take your time, sweetheart. I'm right here.";
          const mod = parseModule(reply);
          if (mod === "SUMMARY") { setSummaryText(reply); setCurrentModule(7); }
          else if (mod) setCurrentModule(parseInt(mod));
          setMessages(prev => [...prev, { role: "assistant", content: reply }]);
        } catch(e) {
          setMessages(prev => [...prev, { role: "assistant", content: "Something interrupted us — please try again." }]);
        }
        setLoading(false);
      };

      const handleChoice = (choice, msgIndex) => {
        const key = `${msgIndex}-${choice}`;
        if (usedChoices.has(key)) return;
        setUsedChoices(prev => new Set([...prev, key]));
        sendMessage(choice);
      };

      const handleKey = e => { if (e.key === "Enter" && !e.shiftKey) { e.preventDefault(); sendMessage(input); } };

      const renderText = (text) => {
        const clean = stripModuleTag(text).replace(/\[CHOICES:[^\]]+\]/g, "").trimEnd();
        return clean.split("\n").map((line, i) => {
          if (line === "") return <div key={i} style={{ height: "0.4em" }} />;
          if (line.includes("---PHOTOGRAPHER SUMMARY---")) return null;
          const parts = line.split(/(\*\*[^*]+\*\*|\*[^*]+\*)/g).map((p, j) => {
            if (p.startsWith("**") && p.endsWith("**")) return <strong key={j}>{p.slice(2,-2)}</strong>;
            if (p.startsWith("*") && p.endsWith("*")) return <em key={j}>{p.slice(1,-1)}</em>;
            return p;
          });
          return <p key={i} style={{ margin: "0.2em 0", lineHeight: "1.75" }}>{parts}</p>;
        });
      };

      const isSummary = currentModule === 7;

      const ProgressBar = () => (
        <div style={{width:"100%",padding:"12px 20px 0"}}>
          <div style={{display:"flex",alignItems:"center"}}>
            {MODULES.map((mod, i) => (
              <div key={mod.id} style={{flex:1,display:"flex",flexDirection:"column",alignItems:"center",position:"relative"}}>
                {i < MODULES.length-1 && <div style={{position:"absolute",top:"8px",left:"50%",width:"100%",height:"1px",background:currentModule>mod.id?"rgba(193,98,47,0.6)":"rgba(255,255,255,0.07)",transition:"background 0.5s"}}/>}
                <div style={{width:"16px",height:"16px",borderRadius:"50%",background:currentModule>mod.id?"#c1622f":currentModule===mod.id?"rgba(193,98,47,0.3)":"rgba(255,255,255,0.06)",border:currentModule===mod.id?"1px solid #c1622f":currentModule>mod.id?"none":"1px solid rgba(255,255,255,0.1)",display:"flex",alignItems:"center",justifyContent:"center",fontSize:"7px",color:currentModule>mod.id?"#000":currentModule===mod.id?"#c1622f":"rgba(255,255,255,0.2)",zIndex:1,transition:"all 0.4s",flexShrink:0}}>
                  {currentModule>mod.id?"✓":mod.symbol}
                </div>
                <div style={{fontSize:"7px",color:currentModule===mod.id?"#c1622f":currentModule>mod.id?"rgba(193,98,47,0.5)":"rgba(255,255,255,0.15)",marginTop:"4px",textTransform:"uppercase",whiteSpace:"nowrap"}}>{mod.name}</div>
              </div>
            ))}
            <div style={{flex:1,display:"flex",flexDirection:"column",alignItems:"center"}}>
              <div style={{width:"16px",height:"16px",borderRadius:"50%",background:isSummary?"#c1622f":"rgba(255,255,255,0.06)",border:isSummary?"none":"1px solid rgba(255,255,255,0.1)",display:"flex",alignItems:"center",justifyContent:"center",fontSize:"7px",color:isSummary?"#000":"rgba(255,255,255,0.15)",zIndex:1}}>✦</div>
              <div style={{fontSize:"7px",color:isSummary?"#c1622f":"rgba(255,255,255,0.15)",marginTop:"4px",textTransform:"uppercase"}}>Ritual</div>
            </div>
          </div>
        </div>
      );

      if (!entered) return (
        <div style={{minHeight:"100vh",background:"#000",display:"flex",flexDirection:"column",alignItems:"center",justifyContent:"center",fontFamily:"Georgia,serif",padding:"40px 24px",textAlign:"center"}}>
          <div>
            <div style={{fontSize:"9px",letterSpacing:"5px",color:"rgba(193,98,47,0.7)",textTransform:"uppercase",marginBottom:"16px"}}>A Wardrobe Ritual</div>

              onMouseEnter={e=>{e.target.style.background="rgba(193,98,47,0.1)"}}
              onMouseLeave={e=>{e.target.style.background="transparent"}}>
              Enter the Ritual
            </button>
          </div>
        </div>
      );

      return (
        <div style={{minHeight:"100vh",background:"#000",display:"flex",flexDirection:"column",alignItems:"center",fontFamily:"Georgia,serif"}}>
          {showResults && summaryText && <ResultsCard summary={summaryText} onClose={()=>setShowResults(false)}/>}
          {showPhotog && summaryText && <PhotographerSummary summary={summaryText} onClose={()=>setShowPhotog(false)}/>}

          <div style={{width:"100%",padding:"14px 24px 0",textAlign:"center",borderBottom:"1px solid rgba(193,98,47,0.1)",background:"rgba(0,0,0,0.98)",position:"sticky",top:0,zIndex:10}}>
            <div style={{fontSize:"8px",letterSpacing:"5px",color:"rgba(193,98,47,0.45)",textTransform:"uppercase",marginBottom:"3px"}}>A Wardrobe Ritual</div>
            <div style={{fontSize:"18px",color:"#f0e6d8",letterSpacing:"1px",fontWeight:"normal",fontFamily:"Georgia,serif"}}>Aunt Bettie's Closet</div>
            <div style={{fontSize:"10px",color:"rgba(193,98,47,0.5)",fontStyle:"italic",marginTop:"3px",marginBottom:"10px"}}>with your Aunt Bettie</div>
            <ProgressBar/>
          </div>

          <div style={{flex:1,width:"100%",maxWidth:"680px",padding:"28px 20px 16px",display:"flex",flexDirection:"column",gap:"22px"}}>
            {messages.map((m, i) => {
              const { clean, choices } = m.role === "assistant" ? parseChoices(m.content) : { clean: m.content, choices: [] };
              const isLastAssistant = m.role === "assistant" && i === messages.length - 1;
              const showChoices = choices.length > 0 && (isLastAssistant || usedChoices.has(`${i}-${choices[0]}`));
              const choicesUsed = choices.length > 0 && usedChoices.has(`${i}-${choices[0]}`);
              return (
                <div key={i}>
                  <div style={{display:"flex",justifyContent:m.role==="user"?"flex-end":"flex-start"}}>
                    {m.role==="assistant" && <div style={{marginRight:"10px",marginTop:"4px",flexShrink:0,opacity:0.65}}><MoonIcon/></div>}
                    <div style={{maxWidth:"82%",padding:"16px 20px",borderRadius:m.role==="user"?"20px 20px 4px 20px":"4px 20px 20px 20px",background:m.role==="user"?"linear-gradient(135deg,#c1622f,#8b3d1a)":"rgba(255,255,255,0.03)",border:m.role==="user"?"none":"1px solid rgba(193,98,47,0.09)",color:m.role==="user"?"#fdf0e6":"#e8ddd0",fontSize:"15px",lineHeight:"1.75"}}>
                      {renderText(clean)}
                    </div>
                  </div>
                  {showChoices && (
                    <div style={{display:"flex",flexWrap:"wrap",gap:"8px",marginTop:"10px",marginLeft:"34px"}}>
                      {choices.map((choice, ci) => {
                        const key = `${i}-${choice}`;
                        const used = usedChoices.has(key);
                        return (
                          <button key={ci} onClick={()=>handleChoice(choice, i)} disabled={choicesUsed}
                            style={{padding:"9px 16px",border:`1px solid ${used?"rgba(193,98,47,0.6)":"rgba(193,98,47,0.25)"}`,background:used?"rgba(193,98,47,0.15)":"transparent",color:used?"#c1622f":choicesUsed?"rgba(255,255,255,0.15)":"rgba(240,230,216,0.65)",fontSize:"13px",fontFamily:"Georgia,serif",fontStyle:"italic",cursor:choicesUsed?"default":"pointer",borderRadius:"30px",transition:"all 0.2s",opacity:choicesUsed&&!used?0.4:1}}>
                            {choice}
                          </button>
                        );
                      })}
                    </div>
                  )}
                </div>
